#### 5.2 嵌入实现（Python示例）

```python
import json
import openai
from typing import List, Dict
import faiss
import numpy as np

class RAGKnowledgeBase:
    def __init__(self, api_key: str):
        self.client = openai.OpenAI(api_key=api_key)
        self.chunks = []
        self.embeddings = []
        self.index = None
    
    def load_chunks(self, file_path: str):
        """加载知识片段"""
        with open(file_path, 'r', encoding='utf-8') as f:
            self.chunks = json.load(f)
        return len(self.chunks)
    
    def embed_chunks(self, batch_size: int = 100):
        """批量向量化"""
        texts = [chunk['text'] for chunk in self.chunks]
        
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i+batch_size]
            response = self.client.embeddings.create(
                model="text-embedding-3-small",
                input=batch
            )
            batch_embeddings = [item.embedding for item in response.data]
            self.embeddings.extend(batch_embeddings)
        
        return len(self.embeddings)
    
    def build_index(self):
        """构建 FAISS 索引"""
        dim = len(self.embeddings[0])
        self.index = faiss.IndexFlatIP(dim)  # 内积相似度
        
        vectors = np.array(self.embeddings, dtype=np.float32)
        faiss.normalize_L2(vectors)
        self.index.add(vectors)
    
    def search(self, query: str, top_k: int = 3) -> List[Dict]:
        """检索相关片段"""
        # 查询向量化
        response = self.client.embeddings.create(
            model="text-embedding-3-small",
            input=[query]
        )
        query_vector = np.array([response.data[0].embedding], dtype=np.float32)
        faiss.normalize_L2(query_vector)
        
        # 检索
        distances, indices = self.index.search(query_vector, top_k)
        
        # 返回结果
        results = []
        for dist, idx in zip(distances[0], indices[0]):
            if idx < len(self.chunks):
                results.append({
                    'chunk': self.chunks[idx],
                    'similarity': float(dist)
                })
        return results

# 使用示例
kb = RAGKnowledgeBase(api_key="your-openai-key")
kb.load_chunks("knowledge_chunks.json")
kb.embed_chunks()
kb.build_index()
results = kb.search("吴健雄是如何发现宇称不守恒的？")
```

#### 5.3 存储方案对比

| 方案 | 优点 | 缺点 | 推荐度 |
|-----|------|------|--------|
| **FAISS 本地** | 快速、免费、可控 | 需要服务器部署 | ⭐⭐⭐⭐ |
| **云数据库** | 免运维、易扩展 | 查询速度可能慢 | ⭐⭐⭐ |
| **Pinecone** | 托管服务、性能好 | 收费、境外服务 | ⭐⭐⭐ |
| **微信云开发** | 与项目无缝集成 | 向量检索需自建 | ⭐⭐⭐ |

**推荐**：先用 **FAISS 本地** 开发调试，部署时转为 **云数据库** 方案。

#### 7.5 敏感词过滤

```python
class SensitiveWordFilter:
    def __init__(self, word_list_path: str):
        with open(word_list_path, 'r', encoding='utf-8') as f:
            self.words = [line.strip() for line in f if line.strip()]
    
    def check(self, text: str) -> bool:
        """检查是否包含敏感词"""
        for word in self.words:
            if word in text:
                return False
        return True
    
    def filter(self, text: str) -> str:
        """过滤敏感词"""
        result = text
        for word in self.words:
            result = result.replace(word, "***")
        return result
```

```python
from aip import AipSpeech

class TTSService:
    def __init__(self, app_id: str, api_key: str, secret_key: str):
        self.client = AipSpeech(app_id, api_key, secret_key)
        # 音色选择：度小美(0)、度小宇(1)、度逍遥(3)、度丫丫(4)
        # 推荐：度小宇(1) - 温和亲切
        self.permalink = 1
    
    def synthesize(self, text: str, output_path: str):
        """合成语音"""
        result = self.client.synthesis(
            text,
            'zh',  # 中文
            1,      # 客户端类型
            {
                'vol': 5,       # 音量 (0-15)
                'per': self.permalink,  # 音色
                'spd': 4,       # 语速 (0-9)，4为正常
                'pit': 5,       # 音调 (0-9)，5为正常
            }
        )
        
        if not isinstance(result, dict):
            with open(output_path, 'wb') as f:
                f.write(result)
            return True
        else:
            print(f"TTS错误: {result}")
            return False
    
    def batch_synthesize(self, scripts: List[Dict], output_dir: str):
        """批量合成"""
        for script in scripts:
            filename = f"{script['id']}.mp3"
            path = f"{output_dir}/{filename}"
            if self.synthesize(script['text'], path):
                script['audioPath'] = f"audio/{filename}"
                print(f"✅ 合成成功: {filename}")
            else:
                print(f"❌ 合成失败: {filename}")

# 使用
tts = TTSService("app_id", "api_key", "secret_key")
scripts = load_scripts("scripts.json")
tts.batch_synthesize(scripts, "output/audio")
```

## 八、风险与应对

| 风险 | 影响 | 应对方案 |
|-----|------|---------|
| TTS效果不理想 | 用户体验差 | 多平台对比测试，选择最佳音色 |
| 大模型幻觉 | 回答错误 | 严格RAG约束 + 敏感词过滤 + 预设问答 |
| 性能瓶颈 | 加载慢 | SVG压缩 + 音频预加载 + CDN |
| 脚本编写工作量大 | 延期 | 分批编写，先完成Level 1必写脚本 |
| 接口响应慢 | 体验差 | 流式输出 + 本地缓存 + 超时降级 |
